  // BOOKING SYSTEM
  const form = document.getElementById('bookForm');
  const confirmBox = document.getElementById('confirm');
  const submitBtn = document.getElementById('submitBtn');

  form.addEventListener('submit', function(e){
    e.preventDefault();

    const name = document.getElementById('name').value.trim();
    const phone = document.getElementById('phone').value.trim();
    const address = document.getElementById('address').value.trim();
    const bins = document.getElementById('bins').value;
    const frequency = document.getElementById('frequency').value;
    const date = document.getElementById('date').value;
    const notes = document.getElementById('notes').value.trim();

    // Check required information
    if(!name || !phone || !address){
      alert("Please complete your name, phone number, and address.");
      return;
    }

    // Prevent double click
    submitBtn.disabled = true;
    submitBtn.innerHTML = "Sending...";

    // Save booking backup locally
    const booking = {
      name: name,
      phone: phone,
      address: address,
      bins: bins,
      frequency: frequency,
      date: date,
      notes: notes,
      time: new Date().toLocaleString()
    };

    try{
      localStorage.setItem("freshBinBooking", JSON.stringify(booking));
    }catch(err){
      console.warn("Could not save local backup:", err);
    }

    // Create SMS message
    const message =
`Fresh Bin Services Booking Request

Customer: ${name}
Phone: ${phone}
Address: ${address}
Bins: ${bins}
Service: ${frequency}
Preferred Date: ${date || "Not selected"}
Notes: ${notes || "None"}`;

    const smsURL = `sms:7745069916?body=${encodeURIComponent(message)}`;

    // Show confirmation
    confirmBox.style.display = "block";
    confirmBox.innerHTML =
      `✅ Thanks ${name.split(" ")[0]}! Your cleaning request has been prepared.<br><br>
      <a href="${smsURL}" style="display:inline-block;background:#7CB82F;color:#0B1A33;padding:12px 18px;border-radius:30px;font-weight:800;">📩 Send Booking Request</a>`;

    confirmBox.scrollIntoView({ behavior: "smooth", block: "center" });

    // Reset button
    setTimeout(() => {
      submitBtn.disabled = false;
      submitBtn.innerHTML = "📅 Request My Cleaning";
    }, 3000);
  });
</script>
